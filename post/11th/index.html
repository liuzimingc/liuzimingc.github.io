struct DSU {
	int n = 0, tot = 0, fa[N], sz[N], s[N];
	void ins() {
		n++, fa[n] = n, sz[n] = 1;   //插入节点
	}
	int F(int x) {
		return fa[x] == x ? x : F(fa[x]);   //即find查找函数
	}
	void U(int x, int y) { //合并函数
		x = F(x), y = F(y);
		if (x == y) return;
		if (sz[x] < sz[y]) swap(x, y);
		s[++tot] = y, fa[y] = x, sz[x] += sz[y];
	}
	void D() { //删除栈顶边

		int y = s[tot--];
		sz[fa[y]] -= sz[y], fa[y] = y;
	}
	void back(int t = 0) {
		while (tot > t) D();   //删除到只剩t条边
	}
